#include <iostream>
#include <fstream>
#include <deque>
#include <vector>
using namespace std;
int l, n;
deque<int> c;
ifstream fisier("input.txt");
struct nod 
{
    int pre;
};
nod v[100];

int get_min() 
{
    // selectam cel mai mic numar natural care
    // nu se gaseste in K

    bool fr[1000] = {0};
    for (auto& i : c)
        fr[i] = 1;

    for (int i = 0; i <= c.size(); i++)
        if (fr[i] == 0) return i;
    return -1;
}

void decodare() 
{
    for (int i = 0; i < l; i++) 
    {
        // luam x = primul element din K
        // luam y = cel mai mic nr natural care nu se gaseste in K

        int y = get_min();
        int x = c.front();
        v[y].pre = x; // adaugam muchia (x,y), unde x este tatal lui y
        c.pop_front(); // stergem x din K
        c.push_back(y); // adaugam y in coada
    }
}

int main() 
{
    // decodare prufer
    fisier >> l;
    n = l + 1;
    for (int i = 0; i < l; i++) 
    {
        int x;
        fisier >> x;
        c.push_back(x);
    }

    for (int i = 0; i < n; i++)
        v[i].pre = -1;

    decodare();

    cout << n << endl;
    for (int i = 0; i < n; i++)
        cout << v[i].pre << " ";

    return 0;
}
